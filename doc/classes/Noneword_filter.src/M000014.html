        def control(cmd, par)
                case cmd
                        when STR_CMD_FILE
                                @nonewords.clear
                        when STR_CMD_EOL
                                deleteCmd
                        when STR_CMD_RECORD, STR_CMD_EOF
                                nones = @nonewords.sort.uniq
                                nones.each { |nw| forward(nw) }
                                add('Objekte gefiltert', nones.size)
                                @nonewords.clear
                end
        end
