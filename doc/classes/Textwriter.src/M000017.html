        def control(cmd, par)
                case cmd
                when STR_CMD_LIR
                        @lir = true
                when STR_CMD_FILE
                        @no_sep = true
                        @filename = par.sub(/(\.[^.]+)?$/, '.'+@ext)
                        @file = File.new(@filename,'w')
                        inc('Anzahl Dateien')
                        
                        @lir_rec_no = ''
                        @lir_rec_buf = Array.new
                        
                when STR_CMD_RECORD
                        @no_sep = true
                        if @lir
                                flush_lir_buffer
                                @lir_rec_no = par
                        end
                        
                when STR_CMD_EOL
                        @no_sep = true
                        unless @lir
                                @file.puts # unless @sep=="\n"

                                inc('Anzahl Zeilen')
                        end
                        
                when STR_CMD_EOF
                        flush_lir_buffer if @lir
                        @file.close
                        add('Anzahl Bytes', File.stat(@filename).size)
                end
        end
