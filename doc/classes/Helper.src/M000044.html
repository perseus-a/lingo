        def init
                case
                        when has_key?('spool_from')
                                @spool_from = get_key('spool_from')
                                @spooler = true
                        when has_key?('dump_to')
                                @dump_to = get_key('dump_to')
                                @spooler = false
                        else
                                forward(STR_CMD_ERR, 'Weder dump_to noch spool_from-Attribut abgegeben')
                end
        end
