        def process(obj)
                if obj.is_a?(String)
                        inc('Anzahl Zeilen')
                        tokenize(obj.strip).each { |token|
                                inc('Anzahl Muster '+token.attr)
                                inc('Anzahl Token')
                                forward(token) 
                        }
                        forward(STR_CMD_EOL, @filename) unless @filename.nil?
                else
                        forward(obj)
                end
        end
