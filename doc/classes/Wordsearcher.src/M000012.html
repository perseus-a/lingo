        def process(obj)
                if obj.is_a?(Token) && obj.attr == TA_WORD
                        inc('Anzahl gesuchter Wörter')
                        word = @dic.find_word(obj.form)
                        inc('Anzahl gefundener Wörter') unless word.attr == WA_UNKNOWN
                        obj = word
                end
                forward(obj)
        end
