        def process(obj)
                if @lir
                        @lir_rec_buf << (obj.kind_of?(Token) ? obj.form : obj.to_s)
                else
                        @file.print @sep unless @no_sep
                        @no_sep=false if @no_sep
                        if obj.is_a?(Word) || obj.is_a?(Token)
                                @file.print obj.form
                        else
                                @file.print obj
                        end
                end
        end
